In [66]:
for col in columns:
    Q3 = X[col].quantile(0.75)
    Q1 = X[col].quantile(0.25)
    IQR = Q3 - Q1
    UB = Q3 + 1.5 * IQR
    LB = Q1 - 1.5 * IQR
    X[col]  =  X[col].apply(lambda x: LB if x < LB else x)
    X[col]  = X[col].apply(lambda x: UB if x > UB else x)
